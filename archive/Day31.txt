        document.write("<br> 배열의 길이 :" + arrr.length+'<br>')
        document.write("전체 배열의 값 : ");

## quiz_5.html

<!DOCTYPE html>
<html>
  <head>
    <meta charset="UTF-8" />
    <title>중복없이 랜덤 숫자 발생</title>
  </head>
  <body>
    <h3>중복없이 랜덤 숫자 발생</h3>
    <hr />
    <script>
      var array = new Array();
      var input = prompt(
        "두개의 숫자(배열의 크기, 최대값)를 입력하세요.(예 5,10)",
        "5,10"
      );
      number = input.split(",");
      for (i = 0; i < number[0]; i++) {
        array[i] = Math.floor(Math.random() * number[1] + 1);
        for (j = 0; j < i; j++) {
          if (array[j] == array[i]) {
            i--;
            continue;
          }
        }
      }
      document.write(`배열의 크기 : ${array.length} <br>`);
      document.write(`입력된 배열 : ${array} <br>`);
    </script>
  </body>
</html>
